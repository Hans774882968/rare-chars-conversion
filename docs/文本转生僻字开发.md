```js
export function buildChineseDict(pinyinDict) {
  const chineseDict = {};

  for (const [key, charList] of Object.entries(pinyinDict)) {
    if (!Array.isArray(charList)) continue;

    for (const s of charList) {
      if (!chineseDict[s]) {
        chineseDict[s] = [key];
      } else {
        chineseDict[s].push(key);
      }
    }
  }

  return chineseDict;
}
```
